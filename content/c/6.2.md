2. 用冒泡法对数组中的10个整数按由小到大排序，要求由键盘任意输入10个数，排序后输出排序结果。

```c
#include <stdio.h>

int main() {
    int a[10] = {};
    for (int i = 0; i < 10; i++) {
        scanf("%d", &a[i]);
    }
    int t;
    for (int i = 1; i <= 10 - 1; i++) {
        for (int j = 0; j < 10 - i; j++) {
            if (a[j] > a[j + 1]) {
                t = a[j];
                a[j] = a[j + 1];
                a[j + 1] = t;
            }
        }
    }
  	return 0;
}
```

3. 输入长度为10的数组， 最大的与第一个元素交换，最小的与最后一个元素交换，输出数组。

```c
#include <stdio.h>

int main() {
    int max = 0, min = 0;

    int a[10] = {};

    for (int i = 0; i < 10; i++) {
        scanf("%d", &a[i]);
        if (a[i] > a[max]) {
            max = i;
        }

        if (a[i] < a[min]) {
            min = i;
        }
    }
    if (a[0] != a[max]) {
        int t = a[0];
        a[0] = a[max];
        a[max] = t;
    }

    if (a[10 - 1] != a[min]) {
        int t = a[10 - 1];
        a[10 - 1] = a[min];
        a[min] = t;
    }

    return 0;
}
```

4. 一个正整数n(1<n<=10)，再输入n个整数，将它们小到大排序后输出。

```c
#include <stdio.h>

int main() {
    int n;
    printf("Enter n: ");
    scanf("%d", &n);
    if (n < 1 || n > 10) {
        printf("请输入一个 1 ~ 10 的整数");
        return 0;
    }

    int a[10] = {};

    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    printf("Enter %d integer: ", n);
    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }

    for (int i = 1; i < n; i++) {
        for (int j = 0; j < n - i; j++) {
            if (a[j] > a[j + 1]) {
                int t = a[j];
                a[j] = a[j + 1];
                a[j + 1] = t;
            }
        }
    }

    printf("\nAfter sorted: ");

    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    return 0;
}
```

6. 编写一个程序，从键盘输入10位学生的成绩， 统计最高分、最低分和平均分。

```c
#include <stdio.h>

int main() {
  int max = 0, min = 0, sum =0;
  int scores[10] = {};
  
  for (int i = 0; i < 10; i++) {
    scanf("%d", &scores[i]);
    if (i == 0) {
      max = scores[0];
      min = scores[0];
    }
    
    if (scores[i] > max) {
      max = scores[i];
    }
    
    if (scores[i] < min) {
      min = scores[i];
    }
    
    sum += scores[i];
  } 
  
  printf("最高分：%d, 最低分：%d, 平均分：%d\n", max, min, sum / 10);
}
```
