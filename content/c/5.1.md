2. 编写程序，求小于正整数k的所有偶数的和 ，其中k由用户从键盘输入。

```c
#include <stdio.h>

int main() {
    unsigned int k = 0, sum = 0;
    scanf("%d", &k);
    for (unsigned int i = k - 1; i != 0; i--) {
        if (i % 2 == 0) {
            sum += i;
        }
    }
    printf("和为：%d", sum);
    return 0;
}
```

输入一个正整数m, 计算 0+2-4+6-8+10-12+...的前m项之和

```c
#include <stdio.h>

int main() {
    unsigned int m = 0, sum = 0;
    scanf("%d", &m);
    for (unsigned int i = 0, j = 0; i < m; i++) {
        if (i % 2 == 0) {
            if (j % 2 == 0) {
                sum += i;
                j++;
            } else {
                sum -= i;
                j++;
            }
        }
    }    
    printf("前 %d 项之和为 %", sum);    
}
```
