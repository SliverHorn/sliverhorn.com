8. 编写函数交换变量a,b的值.

```c
#include <stdio.h>

void swap(int *x, int *y) {
  int t = *x;
  *x = *y;
  *y = t;
}

int main() {
  int x = 0, y = 0;
  scanf("%d", &x);
  scanf("%d", &y);
  swap(&x, &y);
  printf("x = %d, y = %d", x, y);
  return 0;
}
```
