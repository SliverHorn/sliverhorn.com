#include <stdio.h>

int main() {
    int max = 0, min = 0;

    int a[10] = {};

    for (int i = 0; i < 10; i++) {
        scanf("%d", &a[i]);
        if (a[i] > a[max]) {
            max = i;
        }

        if (a[i] < a[min]) {
            min = i;
        }
    }
    if (a[0] != a[max]) {
        int t = a[0];
        a[0] = a[max];
        a[max] = t;
    }

    if (a[10 - 1] != a[min]) {
        int t = a[10 - 1];
        a[10 - 1] = a[min];
        a[min] = t;
    }

    return 0;
}

int main() {
    int n;
    printf("Enter n: ");
    scanf("%d", &n);
    if (n < 1 || n > 10) {
        printf("请输入一个 1 ~ 10 的整数");
        return 0;
    }

    int a[10] = {};

    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    printf("Enter %d integer: ", n);
    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }

    for (int i = 1; i < n; i++) {
        for (int j = 0; j < n - i; j++) {
            if (a[j] > a[j + 1]) {
                int t = a[j];
                a[j] = a[j + 1];
                a[j + 1] = t;
            }
        }
    }

    printf("\nAfter sorted: ");

    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    return 0;
}

#include <stdio.h>

int main() {
  int max = 0, min = 0, sum =0;
  int scores[10] = {};
  
  for (int i = 0; i < 10; i++) {
    scanf("%d", &scores[i]);
    if (i == 0) {
      max = scores[0];
      min = scores[0];
    }
    
    if (scores[i] > max) {
      max = scores[i];
    }
    
    if (scores[i] < min) {
      min = scores[i];
    }
    
    sum += scores[i];
  } 
  
  printf("最高分：%d, 最低分：%d, 平均分：%d\n", max, min, sum / 10);
}
