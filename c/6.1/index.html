    int a[6] = {10,7, 15, 20, 3, 1};
    int max = 0, index = 0;

#include <stdio.h>

int main() {
    int a[9] = {3, 44, 38, 5, 47, 15, 36, 26, 27};

    for (int j = 1; j <= 9 - 1; j++) {
        for (int i = 0; i < 9 - j; i++) {
            if (a[i] > a[i+1]) {
                int t = a[i];
                a[i] = a[i+1];
                a[i+1] = t;
            }
        }
    }

    for (int i = 0; i < 9; i++) {
        printf("%d ", a[i]);
    }
}

#include <stdio.h>
#include <string.h>

int main() {
  char str[20];
  gets(str);
  
  for (int i = 0; str[i] != '\0'; i++) {
    if (str[i] >= 'a' && str[i] <= 'z') {
      str[i] -= 32;
    }
    if (str[i] >= 'A' && str[i] <= 'Z') {
      str[i] += 32;
    }
  }
  puts(str);
  return 0;
}
