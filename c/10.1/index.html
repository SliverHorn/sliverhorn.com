#include <stdio.h>

int main(void) {
    FILE *fp = NULL;
    char ch, filename[10] = {};
    scanf("%s", filename);
    if ((fp = fopen(filename, "w")) == NULL) {
        printf("Cannot open this file\n");
        return 0;
    }
    ch = getchar();
    ch = getchar();
    while (ch != '#') {
        fputc(ch, fp); // 向文件写入字符
        putchar(ch);
        ch = getchar();
    }
    fclose(fp); // 关闭文件
    return 0;
}

    FILE *fp;

    scanf("%s", filename);
    getchar();
    if ((fp = fopen(filename, "r")) == NULL) {
        printf("cannot open file\n");

    while (!feof(fp)) {
        ch = fgetc(fp);
        putchar(ch);
    }
    fclose(fp);
}
