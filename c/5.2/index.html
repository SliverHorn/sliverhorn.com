#include<stdio.h>
#include<math.h>

int main() {
    int n, i, k, count = 0;
    for (n = 101; n <= 999; n += 2) {
        k = (int) sqrt(n);
        for (i = 2; i <= k; i++) {
            if (n % i == 0) {
                break;
            }
            if (i > k) {
                count++;
                printf("%4d", n);
                if (count % 8 == 0) {
                    printf("\n");
                }
            }
        }
    }
    printf("\n");
    return 0;
}

#include <stdio.h>

int main() {
    int sum = 1;
    for (int i = 1, j = 1; i <= 100; i++) {
        if (j % 2 == 0) {
            sum = sum + (1 / (i + 1));
        } else {
            sum = sum - (1 / (i + 1));
        }
    }
    return 0;
}
