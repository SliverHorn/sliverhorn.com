#include <stdio.h>

void swap(int *x, int *y) {
  int temp = 0;
  temp = *x;
  *x = *y;
  *y = temp;
}

int main(void) {
  int a = 10, b = 20;
  swap(&a, &b);
  printf("a = %d, b = %d", a, b);
  return 0;
}

// 输出
// a = 20, b = 10

#include <stdio.h>

int ff(char *str, char ch) {
  int count = 0;
  for (int i = 0; str[i] != '\0'; i++) {
    if (str[i] == ch) {
      count++;
    }
  }
  return count;
}

int main() {
  char string[50] = {}, c;
  scanf("%s", string);
  scanf("%s", &c);
  int n = ff(string, c);
  printf("c = %c 出现次数为 %d \n", c, n);
  return 0;
}
