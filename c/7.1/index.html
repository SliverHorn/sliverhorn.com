#include <stdio.h>

int judge(int x); // 函数的声明

int main() {
  for (int i = 100; i < 999; i++) {
    if (judge(i) == 1) {
      printf("%d ", i);
    }
  }
  return 0;
}

int judge(int x) {
  if (x < 100 || x > 1000) {
    return 0;
  }
  int a = x / 100;
  int b = (x / 10) % 10;
  int c = x % 10;
  if (a * a * a + b * b * b + c * c * c == x) {
    return 1;
  }
  return 0;
}

int prime(int n) {
  for (int i = 0; i < n; i++) {

#include <stdio.h>

long f(int n) {
    long s;
    if (n == 1 || n == 2) {
        return 1;
    }
    s = f(n - 1) + f(n - 2);
    return s;
}

int main(void) {
    int n;
    printf("please input n:");
    scanf("%d", &n);
    if (n < 0) {
        printf("error!\n");
    }
    printf("第%d项的 Fibonacci数列的值为%ld", n, f(n));
    return 0;
}
