long Fibonacci(int n) {
  long s = 0;
  if (n == 1 || n == 2) {
    return 1;
  }
  s = Fibonacci(n - 1) + Fibonacci(n - 2);
  return s
}

  return 0;  
}

#include <stdio.h>

int prime(int n) {
  if (n <= 1) {
    return 0;
  }
  
  for (int i = 2; i < n; i++) {
    if (n % i == 0) {
      return 0;
    }
  }
  return 1;
}

int main () {
  int n = 0;
  scanf("%d", &n);
  int result = prime(n);
  if (result == 0) {
    printf("n = %d， 不是素数", n);
  } else {
    printf("n = %d， 是素数", n);
  }
  return 0;
}

int f(int a, int b) {
    if (a <= 0 && b <= 0) {
        return 0;

    int z = 1;

        z *= a;

    return z;
